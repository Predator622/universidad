#### Pregunta 1: Explique las tablas y que relación existen entre ellas.

```sql
-- CTE para describir la estructura de las tablas
WITH describe_tables AS (
    SELECT 'country' AS table_name, DESCRIBE country FROM DUAL UNION ALL
    SELECT 'city', DESCRIBE city FROM DUAL UNION ALL
    SELECT 'countrylanguage', DESCRIBE countrylanguage FROM DUAL
)
SELECT * FROM describe_tables;
```

#### Pregunta 2: Búsqueda básica

```sql
-- CTE para encontrar diferentes tipos de gobierno sin repetir
WITH distinct_governments AS (
    SELECT DISTINCT governmentform FROM country
)
SELECT * FROM distinct_governments;

-- CTE para encontrar ciudades en Australia
WITH australian_cities AS (
    SELECT c.name, c.district 
    FROM city c
    JOIN country co ON c.countrycode = co.code
    WHERE co.name = 'Australia'
)
SELECT * FROM australian_cities;

-- CTE para países con población mayor a 30 millones ordenados de menor a mayor
WITH high_population_countries AS (
    SELECT name, population 
    FROM country 
    WHERE population > 30000000 
    ORDER BY population ASC
)
SELECT * FROM high_population_countries;

-- CTE para países con área superior a 1 millón y población menor a 20 millones
WITH large_area_small_pop_countries AS (
    SELECT name, surfacearea, population 
    FROM country 
    WHERE surfacearea > 1000000 AND population < 20000000
)
SELECT * FROM large_area_small_pop_countries;
```

#### Pregunta 3: Búsqueda de Strings

```sql
-- CTE para ciudades que tengan solamente una 'a' en su nombre y más largas que 22 caracteres
WITH single_a_cities AS (
    SELECT name 
    FROM city 
    WHERE REGEXP_LIKE(name, '^([^a]*a[^a]*){1}$') AND LENGTH(name) > 22
)
SELECT * FROM single_a_cities;

-- CTE para ciudades con al menos dos 'a' y sin 'e', exactamente 8 caracteres
WITH double_a_no_e AS (
    SELECT c.name, co.name AS country_name 
    FROM city c
    JOIN country co ON c.countrycode = co.code
    WHERE REGEXP_LIKE(c.name, '^([^ae]*a[^ae]*){2}$') AND LENGTH(c.name) = 8
)
SELECT * FROM double_a_no_e;

-- CTE para ciudades con sólo un tipo de vocal en su nombre
WITH single_vowel_cities AS (
    SELECT c.name 
    FROM city c
    WHERE (REGEXP_LIKE(c.name, '^[aeiou]*$') OR REGEXP_LIKE(c.name, '^[bcdfghjklmnpqrstvwxyz]*$'))
)
SELECT * FROM single_vowel_cities;
```

#### Pregunta 4: Búsqueda con condiciones numéricas

```sql
-- CTE para los 5 países más densamente poblados
WITH dense_countries AS (
    SELECT name, population / surfacearea AS density 
    FROM country 
    ORDER BY density DESC 
    LIMIT 5
)
SELECT * FROM dense_countries;

-- CTE para países con superficie menor a 1/5 del promedio
WITH small_surface_countries AS (
    WITH avg_surface AS (SELECT AVG(surfacearea) AS average_surface FROM country)
    SELECT name, surfacearea 
    FROM country c, avg_surface av 
    WHERE surfacearea < (av.average_surface / 5)
)
SELECT * FROM small_surface_countries;

-- CTE para promedio de la expectativa de vida mundial
WITH life_expectancy_avg AS (
    SELECT AVG(lifeexpectancy) AS average_life_expectancy 
    FROM country 
    WHERE lifeexpectancy IS NOT NULL
)
SELECT * FROM life_expectancy_avg;

-- CTE para número de países por continente y promedio de población
WITH continent_population_stats AS (
    SELECT continent, COUNT(name) AS num_countries, AVG(population) AS avg_population 
    FROM country 
    GROUP BY continent 
    ORDER BY continent ASC
)
SELECT * FROM continent_population_stats;

-- CTE para continentes con al menos un país con población superior a 100 millones
WITH high_pop_continent AS (
    SELECT continent, COUNT(name) AS num_countries, SUM(population) AS total_population 
    FROM country 
    WHERE population > 100000000 
    GROUP BY continent
)
SELECT * FROM high_pop_continent;
```

#### Pregunta 5: Búsqueda en más de una tabla

```sql
-- CTE para promedio entre población conocida y calculada por ciudades
WITH avg_population AS (
    SELECT c.name, (c.population + SUM(co.population)) / 2 AS avg_population 
    FROM country c
    LEFT JOIN city co ON c.code = co.countrycode
    GROUP BY c.name, c.population
)
SELECT * FROM avg_population;

-- CTE para países donde se hable inglés y sea oficial
WITH english_official_countries AS (
    SELECT c.name 
    FROM country c
    JOIN countrylanguage cl ON c.code = cl.countrycode
    WHERE cl.language = 'English' AND isofficial = 'T'
)
SELECT * FROM english_official_countries;

-- CTE para nombre de país, idioma oficial, número total de ciudades y ciudad más poblada
WITH city_stats AS (
    SELECT c.name AS country_name, cl.language AS official_language, COUNT(ci.name) AS num_cities, MAX(ci.population) AS max_city_population 
    FROM country c
    JOIN countrylanguage cl ON c.code = cl.countrycode
    LEFT JOIN city ci ON c.code = ci.countrycode
    WHERE cl.isofficial = 'T' AND COUNT(ci.name) >= 5
    GROUP BY c.name, cl.language
)
SELECT * FROM city_stats;

-- CTE para países con idioma oficial hablado por menos del 30% de la población y ciudad más poblada > 50%
WITH language_population AS (
    SELECT c.name AS country_name, cl.language AS official_language, ci.name AS most_populated_city 
    FROM country c
    JOIN countrylanguage cl ON c.code = cl.countrycode
    JOIN city ci ON c.code = ci.countrycode
    WHERE cl.isofficial = 'T' AND (cl.percentage < 30 OR ci.population > c.population * 0.5)
)
SELECT * FROM language_population;
```
