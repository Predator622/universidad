// taller_4_grafos.cxx

#include <iostream>
#include <fstream>
#include <vector>
#include <cmath>
#include <limits>

using namespace std;

struct Point {
    double x, y, z;
};

class Graph {
public:
    struct Edge {
        int from, to;
        double cost;
    };

    vector<vector<Edge>> adjList; // Lista de adyacencia

    void addVertex(int vertex) {
        if (vertex >= adjList.size()) {
            adjList.resize(vertex + 1);
        }
    }

    void addEdge(int u, int v, double cost) {
        adjList[u].push_back({u, v, cost});
        adjList[v].push_back({v, u, cost}); // Grafo no dirigido
    }

    vector<int> dijkstra(int start, int end) const {
        vector<double> dist(adjList.size(), numeric_limits<double>::max());
        vector<bool> visited(adjList.size(), false);
        vector<int> prev(adjList.size(), -1);

        dist[start] = 0;

        for (int i = 0; i < adjList.size(); ++i) {
            int u = min_element(dist.begin(), dist.end()) - dist.begin();
            if (visited[u]) continue;
            visited[u] = true;

            for (const auto& edge : adjList[u]) {
                double newDist = dist[u] + edge.cost;
                if (!visited[edge.to] && newDist < dist[edge.to]) {
                    dist[edge.to] = newDist;
                    prev[edge.to] = u;
                }
            }
        }

        vector<int> path;
        for (int at = end; at != -1; at = prev[at])
            path.push_back(at);
        reverse(path.begin(), path.end());

        return path;
    }
};

void readMesh(const string& filename, Graph& graph) {
    ifstream file(filename);
    int N, E;
    file >> N;

    vector<Point> points(N);

    for (int i = 0; i < N; ++i) {
        file >> points[i].x >> points[i].y >> points[i].z;
        graph.addVertex(i); // Agregar vértice al grafo
    }

    file >> E;

    for (int j = 0; j < E; ++j) {
        int s, e;
        file >> s >> e;
        double cost = sqrt(pow(points[s].x - points[e].x, 2) +
                           pow(points[s].y - points[e].y, 2) +
                           pow(points[s].z - points[e].z, 2));
        graph.addEdge(s, e, cost); // Agregar arista al grafo
    }
}

int main(int argc, char* argv[]) {
    if (argc != 4) return EXIT_FAILURE;

    string filename = argv[1];
    int startVertex = stoi(argv[2]);
    int endVertex = stoi(argv[3]);

    Graph graph;
    readMesh(filename, graph);

    vector<int> path = graph.dijkstra(startVertex, endVertex);
    cout << "Cantidad de vértices en la ruta: " << path.size() - 1 << endl;

    for (int vertex : path) {
        const Point& point = points[vertex];
        cout << "(" << point.x << ", " << point.y << ", " << point.z << ")" << endl;
    }

    return EXIT_SUCCESS;
}
